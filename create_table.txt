create table tb_customer(
id int not null auto_increment,
cust_num VARCHAR(20),
pwd VARCHAR(20),
show_name VARCHAR(100),
true_name VARCHAR(100),
register_time VARCHAR(100),
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;


create table tb_goods(
id int not null auto_increment,
name VARCHAR(100),
img_path VARCHAR(100),
description VARCHAR(100),
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;


create table tb_cart(
id int not null auto_increment,
cust_num int,
goods_id int,
buy_num int,
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;


create table tb_order(
id int not null auto_increment,
cust_num int,
order_time varchar(20),
total_money float,
save_money float,
state smallint,
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;


create table tb_order_detail(
id int not null auto_increment,
order_id int,
goods_id int,
order_num int,
price float,
money float,
save_money float,
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;

create table tb_store(
id int not null auto_increment,
goods_id int,
store_num int,
PRIMARY key (id)
) charset=utf8 ENGINE = INNODB;
